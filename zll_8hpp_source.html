376template < typename T, typename Acc, typename Compare = std::less<> >
377requires( _provides_ll_header< T, Acc > )
378std::pair< T*, T* >
379merge_ranges( T& lhf, T& lhl, T& rhf, T& rhl, Compare&& comp = std::less<>{} ) noexcept(
380 _nothrow_access< Acc, T > && noexcept( comp( lhf, rhf ) ) )
381{
382 detach_range( rhf, rhl );
383 T* lh = &lhf;
384 T* rh = &rhf;
385 T* first = nullptr;
386 T* last = nullptr;
387 _ll_ptr pred = Acc::get( lhf ).prev;
388 _ll_ptr succ = Acc::get( lhl ).next;
389 while ( lh && rh ) {
390 T* tmp = nullptr;
391 if ( comp( *rh, *lh ) ) {
392 tmp = rh;
393 rh = _node( Acc::get( *rh ).next );
394 } else {
395 tmp = lh;
396 if ( lh == &lhl )
397 lh = nullptr;
398 else
399 lh = _node( Acc::get( *lh ).next );
400 }
401 detach( *tmp );
402 if ( !first )
403 first = tmp;
404 if ( last )
405 link_detached_as_next( *last, *tmp );
406 last = tmp;
407 }
408 ZLL_ASSERT( first );
409 ZLL_ASSERT( last );
410 _next_or_first_set< T, Acc >( pred, *first );
411 Acc::get( *first ).prev = pred;
412
413 if ( lh ) {
414 Acc::get( *last ).next = *lh;
415 Acc::get( *lh ).prev = *last;
416 last = &lhl;
417 } else if ( rh ) {
418 Acc::get( *last ).next = *rh;
419 Acc::get( *rh ).prev = *last;
420 last = &rhl;
421
422 Acc::get( *last ).next = succ;
423 _prev_or_last_set< T, Acc >( succ, *last );
424 }
425
426 return { first, last };
